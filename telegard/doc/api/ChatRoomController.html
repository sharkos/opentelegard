def menu
  done = false
  validkeys=['/G']
  until done == true
    key = Tgio::Input.menuprompt('menu_chatroom.ftl',validkeys, nil) # nil is for tpl vars hash
    print "\n"
    case key
      when "/G"
        puts "Goodbye!"
        done = true
        return 0
    end #/case
  end #/until
end
